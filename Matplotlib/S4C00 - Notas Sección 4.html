import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter

plt.style.use("seaborn-whitegrid")

#figura
fig, ax = plt.subplots(figsize = (15,8))

#gráfica
"""
marker -> ['o', '.', ',', 'x', '+', 'v', '^', '<', '>', 's', 'd']
"""

ax.plot(datos.total_fb_likes, datos.imdb_calificacion, marker = "o", linestyle = "", markersize = 9,
       color = "#F0A07C", markeredgecolor = "#4A274F", markeredgewidth = 1.5) #vuelvo transparentes las lineas!

#etiquetas
ax.set_title("Críticas y Popularidad", fontname = "fantasy", fontsize = 35, va = "bottom")
ax.set_xlabel("Likes en Facebook", fontname = "fantasy", fontsize = 20)
ax.set_ylabel("Calificación en imdb", fontname = "fantasy", fontsize = 20)

#ticks
def formato(x, pos):
    return "%.0f mil likes"%(x/1000)

ax.set_xticklabels(list(ax.get_xticks()), fontname = "fantasy", fontsize = 13)
ax.xaxis.set_major_formatter(FuncFormatter(formato))

ax.set_yticklabels(list(ax.get_yticks()), fontname = "fantasy", fontsize = 13)


plt.show()

import matplotlib as mpl
mpl.rcParams.update(mpl.rcParamsDefault)
from matplotlib.ticker import FuncFormatter

plt.style.use("bmh")

#figura
fig3, ax3 = plt.subplots(figsize = (15,8))

#grafica
graf = ax3.scatter(datos.total_fb_likes, datos.imdb_calificacion, marker = "o", s = datos.presupuesto/100000,
           c = datos.ganancia/100000, cmap = "ocean",edgecolor = "black", linewidth = 1.5, alpha = 0.65)

#barra del color
cbar = fig3.colorbar(graf)
cbar.set_label("Ganancias (millones)", fontname = "fantasy")
cbar.ax.set_yticklabels(cbar.get_ticks()/10)

#etiquetas
ax3.set_title("Críticas y Popularidad", fontname = "fantasy", fontsize = 35, va = "bottom")
ax3.set_xlabel("Likes en Facebook", fontname = "fantasy", fontsize = 20)
ax3.set_ylabel("Calificación en imdb", fontname = "fantasy", fontsize = 20)

#ticks
def formato(x, pos):
    return "%.0f mil likes"%(x/1000)

ax3.set_xticklabels(list(ax3.get_xticks()), fontname = "fantasy", fontsize = 13)
ax3.xaxis.set_major_formatter(FuncFormatter(formato))

ax3.set_yticklabels(list(ax3.get_yticks()), fontname = "fantasy", fontsize = 13)

plt.show()

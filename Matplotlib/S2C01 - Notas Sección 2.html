import matplotlib.pyplot as plt
from matplotlib.ticker import AutoMinorLocator, FuncFormatter

%matplotlib inline

#grafica
fig, ax = plt.subplots(figsize = (15,8))

ax.plot(df['Mundo'], lw = 2, color = (25/255,25/255,112/255))

#cuadricula
ax.grid(linestyle = '--', alpha = 0.5)

#quitar spines
ax.spines['top'].set_alpha(0.0) #recuerda que el parametro alpha es sinonimo de transparencia
ax.spines['right'].set_alpha(0.0)
ax.spines['left'].set_alpha(0.5)
ax.spines['bottom'].set_alpha(0.5)

#ticks
ax.tick_params(axis = 'y', which = 'major', length = 10, color = (0,0,0,0.5), direction = 'inout', 
               labelrotation = 45)
ax.set_yticklabels(ax.get_yticklabels(), va = 'center')

ax.tick_params(axis = 'x', which = 'major', length = 10, color = (0,0,0,0.5), direction = 'inout')

ax.xaxis.set_minor_locator(AutoMinorLocator(2))
ax.yaxis.set_minor_locator(AutoMinorLocator(2))

ax.tick_params(axis = "both", which = 'minor', direction = 'in', length = 5, color = (0,0,0,0.5))

#formato de y ticks
def formato(x, pos):
    return "%.1f bn"%(x/1000000000) #los datos estan en billones, divido por un billón para que sea mas facíl leerlo

ax.yaxis.set_major_formatter(FuncFormatter(formato))

#titulo
ax.set_title("Población", fontsize = 30, fontstyle = 'italic', fontname = 'Times New Roman')
ax.set_xlabel("Año", fontsize = 15, fontname = "Times New Roman")
ax.set_ylabel("Cantidad de personas", fontsize = 15, fontname = "Times New Roman")

#texto
ax.text(df.index[-1], df['Mundo'].iloc[-1],
       "Hoy tenemos \n %.2f millones"%(df['Mundo'].iloc[-1]/1000000),
       fontname = "Times New Roman", fontsize = 13, color = (61/255,89/255,171/255), fontstyle = 'italic')

plt.show()

zonas = ['Latam','Europa','USA'] #zonas disponibles

df['Otros'] = df['Mundo'] - df[zonas].sum(axis = 1) # calculo la suma del resto de las zonas

#ultimo año
zonas.append('Otros')
datos = df[zonas].iloc[-1] #recuerda que el -1 es para referirnos al último dato

#grafica
fig3, ax3 = plt.subplots(figsize = (14,14))

colores = ["#63B8FF","#EF5455","#FAD744","#8BD8BD"]
exp = (0.0,0.0,0.2,0.0)

#gráfica de pastel
ax3.pie(datos, labels = zonas, textprops = {"fontsize": 15, "fontname":"Serif", "weight":"bold"},
       colors = colores, shadow = True, explode = exp, autopct = "%.2f%%", pctdistance = 0.8,
       startangle = 150, wedgeprops = {'linewidth':3})

fig3.suptitle("Distribución de Población", fontsize = 25, fontname = "Serif") #título de TODA la figura
ax3.set_title("Año {}".format(df.index[-1]), fontsize = 25, fontname = "Serif") #título de los axes

plt.show()
